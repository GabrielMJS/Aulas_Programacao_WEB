double baseMaior,  baseMenor, altura, areaTrapezio;


Console.WriteLine("Diga a Base Maior");
baseMaior = Convert.ToDouble(Console.ReadLine());

Console.WriteLine("Diga a Base Menor");
baseMenor = Convert.ToDouble(Console.ReadLine());

Console.WriteLine("Diga a Altura");
altura = Convert.ToDouble(Console.ReadLine());



areaTrapezio = (baseMaior + baseMenor) * altura/2;


Console.WriteLine($"A área do trapezio que tem bases " +
    $"{baseMaior} e {baseMenor} e altura {altura} é igual a {areaTrapezio}");

{
    resultado = $"Essa pessoa que tem a idade {idade} não pode votar!";
}
else if ((idade > 16) && (idade < 17))
{
    resultado = $"Essa pessoa que tem a idade {idade} pode votar, seu voto é facultativo!";

}
else if ((idade > 17) && (idade <= 60))
{
    resultado = $"Essa pessoa que tem a idade {idade} pode votar, seu voto é obrigatório!";
}
else
{
    resultado = $"Essa pessoa que tem a idade {idade} pode votar, seu voto é facultativo!";
}


Console.WriteLine(resultado);

//receba 10 numeros do usuario e armazenar os valorodes divididos em vetores pares e impares e depois mostrar os valores em cada

Int32 numero;
Int32[] pares = new int[10];
Int32[] impares = new int[10];

try
{
    for (int i = 0; i < 10; i++)
    {
        Console.WriteLine($"Digite o {i + 1}º numero: ");
        numero = Convert.ToInt32(Console.ReadLine());

        if (numero % 2 == 0)
        {
            pares[i] = numero;
        }
        else
        {
            impares[i] = numero;
        }

    }

    for (int i = 0; i < 10; i++)
    {
        if (pares[i] != 0)
        {
            Console.WriteLine($"par: {pares[i]}");
        }
    }
    for (int i = 0; i < 10; i++)
    {
        if (impares[i] != 0)
        {
            Console.WriteLine($"impar: {impares[i]}");
        }
    }
}
catch (Exception ex)
{
    Console.WriteLine("Digite apenas numeros e maiores que zero para saber se eles são pares ou impares");
}
